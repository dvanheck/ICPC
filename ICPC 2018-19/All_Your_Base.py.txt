import sys

# returns a set of all possible base 10 representations of a string of characters with 0-9 and A-Z
def getPosReps():
    # find minimum possible base

n = sys.stdin.read()

for i in range(n):
    
    a = sys.stdin.read()
    b = sys.stdin.read()

    # convert a and b to all possible base 10 representations and check if they intersect
    aPossReps = getPosReps(a)
    bPossReps = getPosReps(b)
    
    # check if aPossReps and bPossReps intersect
    if aPossReps.intersection(bPossReps):
        print('yes')
